45. 扑克牌顺子

# -*- coding:utf-8 -*-
class Solution:
    def IsContinuous(self, numbers):
        # write code here
        '''1、排序 
        2、如果出现对子，则不是顺子 
        3、计算所有相邻数字间隔总数 
        4、计算0的个数 
        5、如果2、3相等，就是顺子'''
        if not numbers or len(numbers) < 5:
            return False

        numbers.sort()
        for i in range(len(numbers) - 1):
            if numbers[i] == numbers[i + 1] and numbers[i] != 0:
                return False
        split_num = 0
        zeros_num = 0
        for i in range(len(numbers) - 1):
            if numbers[i] == 0:
                zeros_num += 1
            else:
                split_num += numbers[i+1] - numbers[i] - 1
        if split_num == zeros_num or zeros_num == 4:#等于4时，有4个零必能组成顺子
            return True
        else:
            return False
